private fun destroyWebView(webView: WebView) {
  try {
    webView.apply {
      (parent as ViewGroup?)?.removeView(this)
      stopLoading()
      webChromeClient = null
      webViewClient = null
      destroy()
    }
  } catch (e: Exception) {
    logger.error("Something wrong while destroy WebView", e)
  }
}
