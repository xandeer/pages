1. kotlin.math.round

import kotlin.math.roundToInt

fun Double.roundTo(decimals: Int): Double {
  var multiplier = 1.0
  repeat(decimals) { multiplier *= 10 }
  return (this * multiplier).roundToInt() / multiplier
}
