mac daemon

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
<plist version="1.0">
<dict>
  <key>Label</key>
  <string>hs</string>
  <key>UserName</key>
  <string>kevin</string>
  <key>Program</key>
  <string>/Users/kevin/bin/hs-daemon.sh</string>
  <key>RunAtLoad</key>
        <true/>
  <key>StandardErrorPath</key>
  <string>/Users/kevin/temp/hs-log/hs_daemon.stderr</string>
  <key>StandardOutPath</key>
  <string>/Users/kevin/temp/hs-log/hs_daemon.stdout</string>
</dict>
</plist>

# start
sudo launchctl load /Library/LaunchDaemons/hs.plist
# stop
sudo launchctl unload /Library/LaunchDaemons/hs.plist
