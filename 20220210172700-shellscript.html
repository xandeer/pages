2. remove trail slash

3. define function

f() {
  # use `echo` like return
  echo "$1 $2"
}

a=$(f hello sh)

echo $a

4. string contains

a="hello-world"

if [[ $a =~ "-" ]]; then
  echo "a contains '-'"
fi

5. get file name

path=/a/b/c/d.txt
nameWithSuffix="${path##*/}"
echo "with suffix:\t$nameWithSuffix"
nameWithoutSuffix="${nameWithSuffix%.*}"
echo "without suffix:\t$nameWithoutSuffix"
